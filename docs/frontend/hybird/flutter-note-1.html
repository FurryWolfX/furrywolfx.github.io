class Throttle {
  bool _flag = true;
  int delay;

  Throttle({required this.delay});

  void setFlag() {
    _flag = false;
    Future.delayed(Duration(milliseconds: delay)).then((resolve) {
      _flag = true;
    });
  }

  void throttle(void Function() fn) {
    if (_flag == false) {
      return;
    }
    setFlag();
    fn();
  }
}
