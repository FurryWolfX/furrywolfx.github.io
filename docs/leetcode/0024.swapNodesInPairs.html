 * function ListNode(val) {

/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var swapPairs = function (head) {
  const dummy = new ListNode(0);
  dummy.next = head;
  let current = dummy;
  while (current.next != null && current.next.next != null) {
    // åˆå§‹åŒ–åŒæŒ‡é’ˆ

    current = current.next.next;
  }
  return dummy.next;
};

 *     public $val = 0;
 *     public $next = null;

 *         $this->next = $next;

class Solution
{

class Solution {
public:
    ListNode* swapPairs(ListNode* head) {
        ListNode h, *tail = &h;
        while (head && head->next) {
            auto p = head, q = head->next;
            head = q->next;
            q->next = p;
            tail->next = q;
            tail = p;
        }
        tail->next = head;
        return h.next;
    }
};
