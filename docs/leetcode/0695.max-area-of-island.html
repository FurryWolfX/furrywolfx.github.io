class Solution:
    def maxAreaOfIsland(self, grid: List[List[int]]) -> int:
        self.max_area = 0
        self.cnt = 0
        m = len(grid)
        if m == 0: return
        n = len(grid[0])

        def dfs(i, j):
            if i < 0 or i >= m or j < 0 or j >= n: return
            if grid[i][j] == 1:
                self.cnt += 1
                grid[i][j] = 0
                dfs(i + 1, j)
                dfs(i - 1, j)
                dfs(i, j + 1)
                dfs(i, j - 1)

        for i in range(m):
            for j in range(n):
                self.cnt = 0
                dfs(i, j)
                self.max_area = max(self.max_area, self.cnt)
        return self.max_area
